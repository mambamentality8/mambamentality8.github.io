```
#目标: 根据商品分类名称查询品牌列表

#从分类表中拿到name为手机的id
SELECT id FROM tb_category WHERE `name`='手机'

# 通过中间表去拿到手机对应的品牌id
SELECT brand_id FROM tb_category_brand  WHERE category_id in (SELECT id FROM tb_category WHERE `name`='手机')

#最后通过品牌id 拿到品牌名称和品牌id
SELECT `name`,image FROM tb_brand WHERE id in (SELECT brand_id FROM tb_category_brand  WHERE category_id in (SELECT id FROM tb_category WHERE `name`='手机'))
```
